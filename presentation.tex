% --- Slide 1: What is Influence? (ELI5) ---
\begin{frame}{What is Influence? (The 5-Year-Old Explanation)}
    \begin{itemize}
        \item \textbf{Popularity}: Being the loudest kid on the playground. Everyone knows your name, but you're just making noise.
        \item \textbf{Influence}: Being the kid who knows the secret tunnel between the playground and the library.
        \item \textbf{The Difference}: Popularity is about being \textit{seen}. Influence is about being \textit{needed}.
    \end{itemize}
\end{frame}

% --- Slide 2: Popularity vs. Influence ---
\begin{frame}{The "World Record Egg" Paradox}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item \textbf{The Egg}: 60 million likes on Instagram.
                \item \textbf{Result}: Highest popularity in history.
                \item \textbf{The Flaw}: It changed nothing. It had no "structural" impact on how people talk.
                \item \textbf{The Lesson}: Likes are "Vanity Metrics." We want the nodes that change the direction of the conversation.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{Influence is Structural}
                True influence is found in the \textbf{Bridges} (Brokers), not the \textbf{Megaphones} (Popular Hubs).
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 3: Why Reddit? ---
\begin{frame}{Why Reddit? (The Perfect Lab)}
    \begin{itemize}
        \item \textbf{Conversational Trees}: Unlike Twitter (X) or Instagram, Reddit is built on nested hierarchies. Every reply is a clear, directed link.
        \item \textbf{Silos (Subreddits)}: It provides a natural environment of isolated communities, making it perfect for studying "Structural Holes."
        \item \textbf{The API (PRAW)}: High-quality access to historical interaction data, allowing us to reconstruct deep information cascades.
    \end{itemize}
\end{frame}

% --- Slide 4: Objectives ---
\begin{frame}{The Goal: Beyond the Viral Count}
    \begin{itemize}
        \item \textbf{The Core Problem}: Most analysis stops at the "Egg" level (Vanity Metrics).
        \item \textbf{The Missing Link}: Popularity does not equate to structural influence.
        \item \textbf{Our Objective}: Identify the "Shadow Brokers"—the nodes that bridge the gaps between subreddits.
    \end{itemize}
\end{frame}

% --- Slide 2: Data Source: Reddit Hierarchy ---
\begin{frame}{The Anatomy of a Reddit Thread}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{itemize}
                \item \textbf{JSON Structure}: Nested trees where every comment has a `parent_id`.
                \item \textbf{The Bipartite Problem}: Users interact with Threads, but we need User-User interaction.
                \item \textbf{Data Ingestion}: Using PRAW (Python Reddit API Wrapper) to perform "Thread Flattening."
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{block}{The Topology}
                Each reply is an \textbf{Edge}.\\
                Each author is a \textbf{Node}.\\
                Weights are the \textbf{Frequency} of interaction.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% --- Slide 3: The Anti-Fluff Engine ---
\begin{frame}{Cleaning the Noise: Quantile Pruning}
    \begin{itemize}
        \item \textbf{The Signal-to-Noise Ratio}: Social media is 90\% "fluff" (one-off replies).
        \item \textbf{Quantile Strategy}: We calculate the weight distribution of all edges.
        \item \textbf{Pruning}: We discard edges below the 75th percentile (Q3).
        \item \textbf{Benefit}: This statistically isolates the "conversational backbone," reducing a hairball graph into a lean power structure.
    \end{itemize}
\end{frame}

% --- Slide 4: Feature Scaling & Normalization ---
\begin{frame}{Data Science Foundations: Preprocessing}
    \begin{itemize}
        \item \textbf{The Problem}: Nodes with 10k Karma numerically dwarf nodes with 10 Karma.
        \item \textbf{Solution}: `StandardScaler` (Z-score normalization) for all node features.
        \item \textbf{Result}: Ensures that our embedding algorithms (Node2Vec) are not biased toward raw volume, but rather toward structural position.
    \end{itemize}
\end{frame}

% --- Slide 6: The "Crazy" Knowledge: Structural Holes ---
\begin{frame}{Finding the Shadow Brokers}
    \begin{itemize}
        \item \textbf{Burt's Constraint Index}: Measures how much a node's neighbors are already connected to each other.
        \item \textbf{Low Constraint}: Means the node bridges a gap between two silos.
        \item \textbf{The Discovery}: These nodes are the "Gatekeepers." They can halt a cascade or pivot a rumor across subreddit boundaries.
    \end{itemize}
\end{frame}

% --- Slide 7: Network Motifs ---
\begin{frame}{Coordinated Behavior: Motif Analysis}
    \begin{itemize}
        \item \textbf{What are Motifs?}: Recurring 3-node or 4-node subgraph patterns.
        \item \textbf{Patterns}: We look for "Feedbacks" (organic conversation) vs. "Star Patterns" (central coordination).
        \item \textbf{Indicator}: Anomalous star motifs often point to bot activity or coordinated influence campaigns.
    \end{itemize}
\end{frame}

% --- Slide 8: Echo Chambers & Sentiment Polarization ---
\begin{frame}{Polarization: The Sentiment Layer}
    \begin{itemize}
        \item \textbf{Signed Graphs}: Interactions aren't just links; they are (+) or (-).
        \item \textbf{Echo Chamber Index}: We measure the internal sentiment polarity of UMAP clusters.
        \item \textbf{The Finding}: High-density clusters often exhibit "Sentiment Homophily"—where everyone agrees, making the cluster highly resistant to external, contradicting information.
    \end{itemize}
\end{frame}
